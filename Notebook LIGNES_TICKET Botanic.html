from pandas.api.types import is_numeric_dtype
columns=['MONTANTREMISE', 'TOTAL', 'MARGESORTIE']
data7=LGNES_TICKETS[columns]
# Fonction pour dtecter les outliers en utilisant l'IQR
def detect_outliers_iqr(data4, columns):
    Q1 = data7[columns].quantile(0.25)
    Q3 = data7[columns].quantile(0.75)
    IQR = Q3 - Q1
    lower_bound = Q1 - 1.5 * IQR
    upper_bound = Q3 + 1.5 * IQR
    outliers = data7[(data4[columns] < lower_bound) | (data4[columns] > upper_bound)]
    return outliers

# Boucler  travers les colonnes numriques pour dtecter les outliers
for col in data7.columns:
    if is_numeric_dtype(LGNES_TICKETS[col]):
        outliers = detect_outliers_iqr(data7, col)
        if len(outliers) > 0:
            print(f"Outliers dtects dans la colonne '{col}':")
            print(outliers)

from pandas.api.types import is_numeric_dtype
columns=['MONTANTREMISE', 'TOTAL', 'MARGESORTIE']
data7=LGNES_TICKETS[columns]
# Fonction pour imputer les outliers 
def impute_outliers(data7, columns):
    Q1 = np.percentile(data7[columns],25)
    Q3 = np.percentile(data7[columns],75)
    IQR = Q3 - Q1
    lower_bound = Q1 - 1.5 * IQR
    upper_bound = Q3 + 1.5 * IQR
    data7.loc[data7[columns] < lower_bound, columns] = lower_bound 
    data7.loc[data7[columns] > upper_bound,columns]=upper_bound
for columns in columns:
    impute_outliers(data7,columns)
